BScript CHOP — 파이썬으로 만드는 CHOP

Script CHOP은 매 프레임 onCook 함수를 실행해 채널을 직접 만드는 오퍼레이터입니다. 1번 입력에 관절 좌표(select h1:*), 2번 입력에 hand_active를 와이어. 코드는 도킹된 script_fingers_callbacks DAT에:

import math

TIP = {'index': 'index_finger_tip', 'middle': 'middle_finger_tip',
       'ring': 'ring_finger_tip',   'pinky': 'pinky_tip'}
PIP = {'index': 'index_finger_pip', 'middle': 'middle_finger_pip',
       'ring': 'ring_finger_pip',   'pinky': 'pinky_pip'}

R_CURL, R_OPEN = 0.95, 1.25   # 굽힘/펴짐 경계 — 캘리브레이션 노브

def d3(src, a, b):
    # 두 랜드마크 사이 3D 거리 — 07의 Sub→Length 두 노드가 이 세 줄
    return math.sqrt(sum(
        (src['h1:%s:%s' % (a, ax)][0] - src['h1:%s:%s' % (b, ax)][0]) ** 2
        for ax in 'xyz'))

def onCook(scriptOp):
    scriptOp.clear()
    src = scriptOp.inputs[0]
    active = scriptOp.inputs[1]['h1:hand_active'][0]

    ext = {}
    for f in TIP:                                  # 07의 블록 복붙 ×4 = for 루프
        d_tip = d3(src, TIP[f], 'wrist')
        d_pip = d3(src, PIP[f], 'wrist')
        ratio = d_tip / d_pip if d_pip > 1e-6 else 0.0
        ext[f] = max(0.0, min(1.0, (ratio - R_CURL) / (R_OPEN - R_CURL))) * active

    up = [f for f in ext if ext[f] > 0.5]          # 07의 Logic 이진화
    n = len(up)
    rock     = active if n == 0 else 0.0           # 07의 곱셈 3개 = if 세 줄
    paper    = active if n == 4 else 0.0
    scissors = active if (n == 2 and 'index' in up and 'middle' in up) else 0.0

    for name, val in list(ext.items()) + [('count', float(n)),
            ('rock', rock), ('paper', paper), ('scissors', scissors)]:
        c = scriptOp.appendChan(name)              # 채널을 직접 만든다
        c.vals = [val]
